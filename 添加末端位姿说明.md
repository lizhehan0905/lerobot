```py
def hw_to_dataset_features(
    hw_features: dict[str, type | tuple], prefix: str, use_video: bool = True
) -> dict[str, dict]:
    """Convert hardware-specific features to a LeRobot dataset feature dictionary.

    This function takes a dictionary describing hardware outputs (like joint states
    or camera image shapes) and formats it into the standard LeRobot feature
    specification.

    Args:
        hw_features (dict): Dictionary mapping feature names to their type (float for
            joints) or shape (tuple for images).
        prefix (str): The prefix to add to the feature keys (e.g., "observation"
            or "action").
        use_video (bool): If True, image features are marked as "video", otherwise "image".

    Returns:
        dict: A LeRobot features dictionary.
    """
    features = {}

    print("提取的特征")
    for key, value in hw_features.items():
        print(f"  {key}: (type: {type(value)})")

    joint_fts = {
        key: ftype
        for key, ftype in hw_features.items()
        if key.endswith(".pos") or (isinstance(ftype, PolicyFeature) and ftype.type != FeatureType.VISUAL)
    }

    # 添加末端位姿的情况
    endpos_fts = {
        key: ftype
        for key, ftype in hw_features.items()
        if key.endswith(".epos") or (isinstance(ftype, PolicyFeature) and ftype.type != FeatureType.VISUAL)
    }

    cam_fts = {key: shape for key, shape in hw_features.items() if isinstance(shape, tuple)}

    if joint_fts and prefix == ACTION:
        features[prefix] = {
            "dtype": "float32",
            "shape": (len(joint_fts),),
            "names": list(joint_fts),
        }

    # 单一情况
    if joint_fts and prefix == OBS_STR:
        features[f"{prefix}.state"] = {
            "dtype": "float32",
            "shape": (len(joint_fts),),
            "names": list(joint_fts),
        }

    if endpos_fts and prefix == OBS_STR:
        features[f"{prefix}.state_epos"] = {
            "dtype": "float32",   # 注意这里是int32
            "shape": (len(endpos_fts),),
            "names": list(endpos_fts),
        }

    for key, shape in cam_fts.items():
        features[f"{prefix}.images.{key}"] = {
            "dtype": "video" if use_video else "image",
            "shape": shape,
            "names": ["height", "width", "channels"],
        }

    _validate_feature_names(features)
    return features

```

```py
def build_dataset_frame(
    ds_features: dict[str, dict], values: dict[str, Any], prefix: str
) -> dict[str, np.ndarray]:
    """Construct a single data frame from raw values based on dataset features.

    A "frame" is a dictionary containing all the data for a single timestep,
    formatted as numpy arrays according to the feature specification.

    Args:
        ds_features (dict): The LeRobot dataset features dictionary.
        values (dict): A dictionary of raw values from the hardware/environment.
        prefix (str): The prefix to filter features by (e.g., "observation"
            or "action").

    Returns:
        dict: A dictionary representing a single frame of data.
    """
    # print("values字典内容:")
    # for key, value in values.items():
    #     print(f"  {key}: (type: {value})")

    frame = {}
    for key, ft in ds_features.items():
        # print("=======================")
        # print(prefix)
        # print("================== key",key)
        # print("================== ft",ft)
        if key in DEFAULT_FEATURES or not key.startswith(prefix):
            continue
        elif key == "action" and ft["dtype"] == "float32" and len(ft["shape"]) == 1:
            # print("记录action数据")
            frame[key] = np.array([values[name] for name in ft["names"]], dtype=np.float32)

        elif key == "observation.state" and ft["dtype"] == "float32" and len(ft["shape"]) == 1:
            # print("记录state pos数据")
            frame[key] = np.array([values[name] for name in ft["names"]], dtype=np.float32)

        elif key == "observation.state_epos" and ft["dtype"] == "float32" and len(ft["shape"]) == 1:
            # print("记录state_endpos数据")
            frame[key] = np.array([values[name] for name in ft["names"]], dtype=np.float32)

        elif ft["dtype"] in ["image", "video"]:
            # print("记录图像数据")
            frame[key] = values[key.removeprefix(f"{prefix}.images.")]

    return frame
```
